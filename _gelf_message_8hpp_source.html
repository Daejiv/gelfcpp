154 private:
155  void SetField(const std::string& field, rapidjson::Value&& value)
156  {
157  static const std::unordered_set<std::string> BUILDIN_FIELDS{ "version", "host", "short_message", "full_message", "timestamp", "level" };
158 
159  rapidjson::Value key;
160  if (BUILDIN_FIELDS.find(field) != BUILDIN_FIELDS.end())
161  key.SetString(field.c_str(), static_cast<rapidjson::SizeType>(field.size()), doc_.GetAllocator());
162  else
163  key.SetString(("_" + field).c_str(), static_cast<rapidjson::SizeType>(field.size() + 1), doc_.GetAllocator());
164 
165  auto find = doc_.FindMember(key);
166  if (find != doc_.MemberEnd())
167  find->value = std::move(value);
168  else
169  doc_.AddMember(std::move(key), std::move(value), doc_.GetAllocator());
170  }
