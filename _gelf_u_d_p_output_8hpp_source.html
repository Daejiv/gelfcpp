16  *
17  * Serializes and sends messages to e.g. an Graylog Node, sending is non-blocking.
18  *
19  * Uses chunking for big messages due to size limitations of UDP.
20  * Additionally uses compression if enabled (by setting \c GELFCPP_WITH_COMPRESSION in CMake).
21  */
22 class GelfUDPOutput
23 {
24 public:
25  /**
26  * \brief Creates an new output socket to given host and port
27  *
28  * \param host remote hostname
29  * \param port remote UDP port
30  */
31  GelfUDPOutput(const std::string& host, uint16_t port)
32  {
33  boost::asio::ip::udp::resolver resolver(service_);
34  boost::asio::ip::udp::resolver::query query(host, std::to_string(port));
35  endpoint_ = *resolver.resolve(query);
36  socket_.reset(new boost::asio::ip::udp::socket(service_, endpoint_.protocol()));
37  }
38 
39  /**
40  * \brief Serializes a message to the UDP stream
41  *
42  * \param message the message
43  */
44  void Write(const GelfMessage& message)
45  {
46  for (const std::string& chunk : serializer_.Serialize(message))
47  socket_->async_send_to(boost::asio::buffer(chunk), endpoint_, [](const boost::system::error_code&, unsigned long int) {});
48  }
49 
50 private:
